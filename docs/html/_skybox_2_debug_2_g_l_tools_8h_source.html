94 // Macros for big/little endian happiness
95 // These are intentionally written to be easy to understand what they
96 // are doing... no flames please on the inefficiency of these.
97 #ifdef __BIG_ENDIAN__
98 // This function says, "this pointer is a little endian value"
100 // If the value must be changed it is... otherwise, this
101 // function is defined away below (on Intel systems for example)
102 inline void LITTLE_ENDIAN_WORD(void *pWord)
103  {
104  unsigned char *pBytes = (unsigned char *)pWord;
105  unsigned char temp;
106 
107  temp = pBytes[0];
108  pBytes[0] = pBytes[1];
109  pBytes[1] = temp;
110  }
111 
113 // This function says, "this pointer is a little endian value"
114 // If the value must be changed it is... otherwise, this
115 // function is defined away below (on Intel systems for example)
116 inline void LITTLE_ENDIAN_DWORD(void *pWord)
117  {
118  unsigned char *pBytes = (unsigned char *)pWord;
119  unsigned char temp;
120 
121  // Swap outer bytes
122  temp = pBytes[3];
123  pBytes[3] = pBytes[0];
124  pBytes[0] = temp;
125 
126  // Swap inner bytes
127  temp = pBytes[1];
128  pBytes[1] = pBytes[2];
129  pBytes[2] = temp;
130  }
131 #else
132 
133 // Define them away on little endian systems
134 #define LITTLE_ENDIAN_WORD
135 #define LITTLE_ENDIAN_DWORD
136 #endif
