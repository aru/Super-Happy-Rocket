93  // Calculates the corners of the Frustum and sets the projection matrix.
94  // Perspective Matrix Projection
95  void SetPerspective(float fFov, float fAspect, float fNear, float fFar)
96  {
97  float xmin, xmax, ymin, ymax; // Dimensions of near clipping plane
98  float xFmin, xFmax, yFmin, yFmax; // Dimensions of far clipping plane
99 
100  // Do the Math for the near clipping plane
101  ymax = fNear * float(tan( fFov * M3D_PI / 360.0 ));
102  ymin = -ymax;
103  xmin = ymin * fAspect;
104  xmax = -xmin;
105 
106  // Construct the projection matrix
107  m3dLoadIdentity44(projMatrix);
108  projMatrix[0] = (2.0f * fNear)/(xmax - xmin);
109  projMatrix[5] = (2.0f * fNear)/(ymax - ymin);
110  projMatrix[8] = (xmax + xmin) / (xmax - xmin);
111  projMatrix[9] = (ymax + ymin) / (ymax - ymin);
112  projMatrix[10] = -((fFar + fNear)/(fFar - fNear));
113  projMatrix[11] = -1.0f;
114  projMatrix[14] = -((2.0f * fFar * fNear)/(fFar - fNear));
115  projMatrix[15] = 0.0f;
116 
117  // Do the Math for the far clipping plane
118  yFmax = fFar * float(tan(fFov * M3D_PI / 360.0));
119  yFmin = -yFmax;
120  xFmin = yFmin * fAspect;
121  xFmax = -xFmin;
122 
123 
124  // Fill in values for untransformed Frustum corners
125  // Near Upper Left
126  nearUL[0] = xmin; nearUL[1] = ymax; nearUL[2] = -fNear; nearUL[3] = 1.0f;
127 
128  // Near Lower Left
129  nearLL[0] = xmin; nearLL[1] = ymin; nearLL[2] = -fNear; nearLL[3] = 1.0f;
130 
131  // Near Upper Right
132  nearUR[0] = xmax; nearUR[1] = ymax; nearUR[2] = -fNear; nearUR[3] = 1.0f;
133 
134  // Near Lower Right
135  nearLR[0] = xmax; nearLR[1] = ymin; nearLR[2] = -fNear; nearLR[3] = 1.0f;
136 
137  // Far Upper Left
138  farUL[0] = xFmin; farUL[1] = yFmax; farUL[2] = -fFar; farUL[3] = 1.0f;
139 
140  // Far Lower Left
141  farLL[0] = xFmin; farLL[1] = yFmin; farLL[2] = -fFar; farLL[3] = 1.0f;
142 
143  // Far Upper Right
144  farUR[0] = xFmax; farUR[1] = yFmax; farUR[2] = -fFar; farUR[3] = 1.0f;
145 
146  // Far Lower Right
147  farLR[0] = xFmax; farLR[1] = yFmin; farLR[2] = -fFar; farLR[3] = 1.0f;
148  }

151  // Builds a transformation matrix and transforms the corners of the Frustum,
152  // then derives the plane equations
153  void Transform(GLFrame& Camera)
154  {
155  // Workspace
156  M3DMatrix44f rotMat;
157  M3DVector3f vForward, vUp, vCross;
158  M3DVector3f vOrigin;
159 
161  // Create the transformation matrix. This was the trickiest part
162  // for me. The default view from OpenGL is down the negative Z
163  // axis. However, building a transformation axis from these
164  // directional vectors points the frustum the wrong direction. So
165  // You must reverse them here, or build the initial frustum
166  // backwards - which to do is purely a matter of taste. I chose to
167  // compensate here to allow better operability with some of my other
168  // legacy code and projects. RSW
169  Camera.GetForwardVector(vForward);
170  vForward[0] = -vForward[0];
171  vForward[1] = -vForward[1];
172  vForward[2] = -vForward[2];
173 
174  Camera.GetUpVector(vUp);
175  Camera.GetOrigin(vOrigin);
176 
177  // Calculate the right side (x) vector
178  m3dCrossProduct3(vCross, vUp, vForward);
179 
180  // The Matrix
181  // X Column
182  memcpy(rotMat, vCross, sizeof(float)*3);
183  rotMat[3] = 0.0f;
184 
185  // Y Column
186  memcpy(&rotMat[4], vUp, sizeof(float)*3);
187  rotMat[7] = 0.0f;
188 
189  // Z Column
190  memcpy(&rotMat[8], vForward, sizeof(float)*3);
191  rotMat[11] = 0.0f;
192 
193  // Translation
194  rotMat[12] = vOrigin[0];
195  rotMat[13] = vOrigin[1];
196  rotMat[14] = vOrigin[2];
197  rotMat[15] = 1.0f;
198 
200  // Transform the frustum corners
201  m3dTransformVector4(nearULT, nearUL, rotMat);
202  m3dTransformVector4(nearLLT, nearLL, rotMat);
203  m3dTransformVector4(nearURT, nearUR, rotMat);
204  m3dTransformVector4(nearLRT, nearLR, rotMat);
205  m3dTransformVector4(farULT, farUL, rotMat);
206  m3dTransformVector4(farLLT, farLL, rotMat);
207  m3dTransformVector4(farURT, farUR, rotMat);
208  m3dTransformVector4(farLRT, farLR, rotMat);
209 
211  // Derive Plane Equations from points... Points given in
212  // counter clockwise order to make normals point inside
213  // the Frustum
214  // Near and Far Planes
215  m3dGetPlaneEquation(nearPlane, nearULT, nearLLT, nearLRT);
216  m3dGetPlaneEquation(farPlane, farULT, farURT, farLRT);
217 
218  // Top and Bottom Planes
219  m3dGetPlaneEquation(topPlane, nearULT, nearURT, farURT);
220  m3dGetPlaneEquation(bottomPlane, nearLLT, farLLT, farLRT);
221 
222  // Left and right planes
223  m3dGetPlaneEquation(leftPlane, nearLLT, nearULT, farULT);
224  m3dGetPlaneEquation(rightPlane, nearLRT, farLRT, farURT);
225  }

240  // Test a point against all frustum planes. A negative distance for any
241  // single plane means it is outside the frustum. The radius value allows
242  // to test for a point (radius = 0), or a sphere. Possibly there might
243  // be some gain in an alternative function that saves the addition of
244  // zero in this case.
245  // Returns false if it is not in the frustum, true if it intersects
246  // the Frustum.
247  bool TestSphere(M3DVector3f vPoint, float fRadius)
248  {
249  float fDist;
250 
251  // Near Plane - See if it is behind me
252  fDist = m3dGetDistanceToPlane(vPoint, nearPlane);
253  if(fDist + fRadius <= 0.0)
254  return false;
255 
256  // Distance to far plane
257  fDist = m3dGetDistanceToPlane(vPoint, farPlane);
258  if(fDist + fRadius <= 0.0)
259  return false;
260 
261  fDist = m3dGetDistanceToPlane(vPoint, leftPlane);
262  if(fDist + fRadius <= 0.0)
263  return false;
264 
265  fDist = m3dGetDistanceToPlane(vPoint, rightPlane);
266  if(fDist + fRadius <= 0.0)
267  return false;
268 
269  fDist = m3dGetDistanceToPlane(vPoint, bottomPlane);
270  if(fDist + fRadius <= 0.0)
271  return false;
272 
273  fDist = m3dGetDistanceToPlane(vPoint, topPlane);
274  if(fDist + fRadius <= 0.0)
275  return false;
276 
277  return true;
278  }
